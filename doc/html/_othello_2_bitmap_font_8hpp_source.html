83 /*
84  * Analiza la superficie ima buscando letras para registrarlas en el
85  * vector de rectángulos fuentes (segundo parámetro)
86  *
87  * Para analizar, el programa recorre la imágen de izquierda a derecha
88  * 'leyendo' barras verticales de píxeles. Así determina donde comienza
89  * y termina cada letra (debe existir una separación de, al menos, un
90  * pixel entre cada caracter)
91  */
92  void identificarRects (){
93  int indice_letra = 0;
94  bool esta_recorriendo_letra = false;
95  int w = 0; /* ancho de la letra actual */
96 
97  for (int i = 0; i < mTexture->getWidth(); i ++)
98  {
99  if (esColumnaVacia ( i))
100  {
101 
102  /* si estaba leyendo una letra entonces encontró
103  * el ancho de la misma */
104  if (esta_recorriendo_letra)
105  {
106  fuentes[indice_letra].x=i - w;
107  fuentes[indice_letra].y=0;
108  fuentes[indice_letra].w=w;
109  fuentes[indice_letra].h=mTexture->getHeight();
110  esta_recorriendo_letra = false;
111  indice_letra ++;
112  }
113  }
114  else
115  {
116  /* si está recorriendo una letra aumenta el contador
117  * de ancho, en caso contrario encontró donde comienza
118  * la siguiente letra
119  */
120  if (esta_recorriendo_letra)
121  w ++;
122  else
123  {
124  w = 1;
125  esta_recorriendo_letra = true;
126  }
127  }
128  }
129 
130  std::cout<<"Econtré " <<indice_letra <<" letras en el archivo de fuentes\n";
