442  void estadoRunningUpdate(){
443 
444  mControlAnimaciones->update();
445 
446  if(mDequeTetraminosCayendo.size() > 0) {
447 
448  mActualDelayBajarTetraminosFlotantes++;
449 
450  if(mActualDelayBajarTetraminosFlotantes >= mMaxDelayBajarTetraminosFlotantes){
451  mActualDelayBajarTetraminosFlotantes = 0;
452 
453  bool almenosUnTetraminoTocoPiso = false;
454 
455  auto itBloqueAnimacionCayendo = mDequeTetraminosCayendo.begin();
456  while (itBloqueAnimacionCayendo != mDequeTetraminosCayendo.end()) {
457 
458  (*itBloqueAnimacionCayendo)->moveIp(0, mSizeCuadro);
459  if (estaTetraminoEnPosicionInvalida((*itBloqueAnimacionCayendo))) {
460  (*itBloqueAnimacionCayendo)->moveIp(0, -mSizeCuadro);
461  guardarTetrominoEnFondo((*itBloqueAnimacionCayendo));
462  itBloqueAnimacionCayendo = mDequeTetraminosCayendo.erase(itBloqueAnimacionCayendo);
463  almenosUnTetraminoTocoPiso = true;
464  } else {
465  ++itBloqueAnimacionCayendo;
466  }
467  }
468 
469  if(almenosUnTetraminoTocoPiso) {
470  int nLineasCompletas = limpiarLineasCompletas();
471  if(nLineasCompletas){
472  mParent->tetrisLineasCompletadas(mId,nLineasCompletas);
473  }
474  }
475 
476  if(mDequeTetraminosCayendo.size() == 0 &&
477  mControlAnimaciones->numeroAnimacionesActivas(GRUPO_ANIMACION_ELIMINAR_LINEAS) == 0) {
478  crearTetrominos();
479  }
480  }
481 
482  }
483 
484  if(mTetrominoActual != nullptr){
485 
486  const Uint8 *teclas= SDL_GetKeyboardState(NULL);//se obtiene el estado_actual actual del teclado
487  mControlPlayerTeclas->update(teclas);
488  //mTetrominoActual->update(nullptr);
489 
490  mDelayActualBajarTetrominoActual++;
491 
492  if(mControlPlayerTeclas->estaTeclaAceptada(ControlTeclasPlayer::TECLA_MOVER_LEFT)){
493  if (moverIpTetromino(mTetrominoActual, -mSizeCuadro, 0)) {
494  mParent->playSfx(mSfxPieceMove);
495  mDelayActualBajarTetrominoActual = 0;
496  } else {
497  mParent->playSfx(mSfxPieceTouch);
498  }
499  }else if(mControlPlayerTeclas->estaTeclaAceptada(ControlTeclasPlayer::TECLA_MOVER_RIGHT)){
500  if (moverIpTetromino(mTetrominoActual, mSizeCuadro, 0)) {
501  mParent->playSfx(mSfxPieceMove);
502  mDelayActualBajarTetrominoActual = 0;
503  } else {
504  mParent->playSfx(mSfxPieceTouch);
505  }
506  }else if(mControlPlayerTeclas->estaTeclaAceptada(ControlTeclasPlayer::TECLA_HARD_DROP)) {
507  int nCeldasDropeadas = hardDropTetramino(mTetrominoActual);
508  if(nCeldasDropeadas > 0){
509  mParent->tetrisHardDrop(mId,nCeldasDropeadas);
510  }
511 
512  mDelayActualBajarTetrominoActual = mMaxDelayBajarTetrominoActual + 1;
513  mParent->playSfx(mSfxHoldPiece);
514  }else if(mControlPlayerTeclas->estaTeclaPresionada(ControlTeclasPlayer::TECLA_SOFT_DROP) &&
515  !mEstaUtilizandoSoftDrop) {
516  mDelayActualBajarTetrominoActual = mMaxDelayBajarTetrominoActual + 1;
517  mMaxDelayBajarTetrominoActual /= 4;
518  mParent->playSfx(mSfxSoftDrop);
519  mCeldasAvanzadasSoftDrop = 0;
520  mEstaUtilizandoSoftDrop = true;
521 
522  }else if(!mControlPlayerTeclas->estaTeclaPresionada(ControlTeclasPlayer::TECLA_SOFT_DROP) &&
523  mEstaUtilizandoSoftDrop) {
524  mMaxDelayBajarTetrominoActual *= 4;
525  mParent->playSfx(mSfxHardDrop);
526  if (mCeldasAvanzadasSoftDrop) {
527  mParent->tetrisSoftDrop(mId, mCeldasAvanzadasSoftDrop);
528  mCeldasAvanzadasSoftDrop = 0;
529  }
530  mEstaUtilizandoSoftDrop = false;
531  }else if(mControlPlayerTeclas->estaTeclaAceptada(ControlTeclasPlayer::TECLA_GIRAR_LEFT)) {
532  if (girarFormaActualDesc(1)) {
533  mDelayActualBajarTetrominoActual = 0;
534  mTetrominoGhost->rotate(1);
535  }
536  }else if(mControlPlayerTeclas->estaTeclaAceptada(ControlTeclasPlayer::TECLA_GIRAR_RIGHT)) {
537  if (girarFormaActualDesc(-1)) {
538  mTetrominoGhost->rotate(-1);
539  mDelayActualBajarTetrominoActual = 0;
540  }
541  }else if(mControlPlayerTeclas->estaTeclaAceptada(ControlTeclasPlayer::TECLA_PAUSA)) {
542  mParent->tetrisPaused(mId);
543  mEstadoJuego = EstadoJuego ::PAUSADO;
544  mOpcionSeleccionadaMenuPausa = 0;
545  mpBitFntRendOpsMenuPausa[mOpcionSeleccionadaMenuPausa]->setBitmapFont(mpBitmapFont[EstadoOpcionMenu::RESALTADO]);
546  }
547 
548  mTetrominoGhost->move(mTetrominoActual->getX(),mTetrominoActual->getY());
549  hardDropTetramino(mTetrominoGhost);
550 
551  if(mDelayActualBajarTetrominoActual >= mMaxDelayBajarTetrominoActual){
552  mDelayActualBajarTetrominoActual = 0;
553
