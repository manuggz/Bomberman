85  void tetrisLineasCompletadas(int TetrisID, int nLineasCompletadas) override {
86 
87  switch(nLineasCompletadas){
88  case SINGLE_LINE:
89  mPuntajePlayer += 100*mLevelTetrisPlayer;
90  break;
91  case DOUBLE_LINE:
92  mPuntajePlayer += 300*mLevelTetrisPlayer;
93  break;
94  case TRIPLE_LINE:
95  mPuntajePlayer += 500*mLevelTetrisPlayer;
96  break;
97  case TETRIS:
98  if(mLineasCompletasAnteriores == TETRIS){
99  mPuntajePlayer += 1200 * mLevelTetrisPlayer;
100  mGameManagerInterfaz->play(mSfxBTBTetris);
101  }else{
102  mPuntajePlayer += 800*mLevelTetrisPlayer;
103  }
104  break;
105  default:break; // Help to avoid warnings!
106  }
107 
108  if(mLineasCompletasAnteriores){
109  mPuntajePlayer += 50*mLineasCompletasAnteriores*mLevelTetrisPlayer;
110 
111  if(mLineasCompletas >= DOUBLE_LINE && mLineasCompletasAnteriores >= DOUBLE_LINE){
112  if(rand()%3 == 1){
113  mGameManagerInterfaz->play(mSfxCongratulate[rand()%8]);
114  }
115  }
116  }
117  mLineasCompletasAnteriores = nLineasCompletadas;
118 
119 
120  if(mPuntajePlayer > mHighScore && mHighScore > 0&&!mCongratuledScoreGreaterThanHighScore){
121  mCongratuledScoreGreaterThanHighScore = true;
122  mGameManagerInterfaz->play(mSfxCongratulate[rand()%8]);
123  mBitmapScorePlayer1Valor->setBitmapFont(mBitmapFont[RESALTADO]);
124  }
125 
126  setTextWithDigits(mBitmapScorePlayer1Valor,mPuntajePlayer,10);
127 
128  mLineasCompletas += nLineasCompletadas;
129  if((mLineasCompletas / 10 + 1) > mLevelTetrisPlayer){
130  int nuevoTick = mTetrisJuego->getTickDelayBajarTetromino() - 2;
131  if(nuevoTick <= 2){
132  nuevoTick = 2;
133  }
134  mTetrisJuego->setTickDelayBajarTetromino(nuevoTick);
135  mLevelTetrisPlayer += 1;
136  setTextWithDigits(mBitmapLevelPlayer1Valor,mLevelTetrisPlayer,3);
137  mGameManagerInterfaz->play(mSfxLevelUp);
138  }
139 
140  setTextWithDigits(mBitmapLevelPlayer1Valor,mLineasCompletas,5);
141 
142  mGameManagerInterfaz->play(mSfxClearLines[nLineasCompletadas - 1]);
143  }
